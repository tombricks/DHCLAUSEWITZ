	DHCW_Politics_scripted_gui = {
		context_type = player_context
		window_name = "DHCW_Politics_scripted_gui"
		parent_window_token = politics_tab

		properties = {
			minister_gfx = {
				image = "GFX_MINISTER_[?v]"
			}
		}

		dynamic_lists = {
			Cabinet_Grid = {
				array = cabinet
				entry_container = DHCW_Politics_Cabinet_Entry
			}
		}

		effects = {
			minister_gfx_click = {
				if = {
					limit = {
						check_variable = { minister_selection = i }
					}
					set_variable = { minister_selection = -1 }
					else = {
						set_variable = { minister_selection = i }
					}
				}
			}
		}
	}
